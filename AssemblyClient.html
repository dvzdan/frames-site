(function() { var state = { steps: [], activeIndex: 0, reference: null }; function text(value) { return value == null ? "" : String(value); } function byOrder(a, b) { return (Number(a && a.order) || 0) - (Number(b && b.order) || 0); } function createElement(tag, className, content) { var node = document.createElement(tag); if (className) node.className = className; if (content != null) node.textContent = text(content); return node; } function getAssemblyManifest() { var manifest = window.ASSEMBLY_MANIFEST || {}; return { version: manifest.version || 1, imageBase: text(manifest.imageBase).replace(/\/$/, ""), reference: manifest.reference || null, sections: Array.isArray(manifest.sections) ? manifest.sections.slice().sort(byOrder) : [] }; } function getManifestImageUrl(manifest, image) { return [ manifest.imageBase, encodeURIComponent(text(image)) ].join("/"); } function getStepImageUrl(manifest, section, step) { return [ manifest.imageBase, encodeURIComponent(text(section.folder)), encodeURIComponent(text(step.image)) ].join("/"); } function getDisplayStepNumber(sectionIndex, step) { var order = Number(step.order); var whole = Math.floor(order); var suffix = ""; if (step.kind === "addendum") { suffix = "a"; } else if (order % 1 !== 0) { suffix = String(order).split(".")[1] || ""; } return String(sectionIndex + 1) + "." + String(whole) + suffix; } function flattenManifest(manifest) { var steps = []; manifest.sections.forEach(function(section, sectionIndex) { var sectionSteps = Array.isArray(section.steps) ? section.steps.slice().sort(byOrder) : []; var parentNumber = ""; var parentTitle = ""; sectionSteps.forEach(function(step) { var isAddendum = step.kind === "addendum"; var number = getDisplayStepNumber(sectionIndex, step); if (!isAddendum) { parentNumber = number; parentTitle = text(step.title); } steps.push({ section: section, sectionIndex: sectionIndex, step: step, number: number, isAddendum: isAddendum, parentNumber: isAddendum ? parentNumber : number, parentTitle: isAddendum ? parentTitle : text(step.title), imageUrl: getStepImageUrl(manifest, section, step) }); }); }); return steps; } function getReference(manifest) { if (!manifest.reference || !manifest.reference.image) return null; return { title: text(manifest.reference.title || "Reference"), imageUrl: getManifestImageUrl(manifest, manifest.reference.image) }; } function createButton(label, className, onClick) { var button = createElement("button", className, label); button.type = "button"; button.addEventListener("click", onClick); return button; } function setActiveStep(index) { if (!state.steps[index]) return; state.activeIndex = index; renderActiveStep(); } function moveStep(delta) { setActiveStep(Math.max(0, Math.min(state.steps.length - 1, state.activeIndex + delta))); } function renderActiveStep() { var root = document.getElementById("assemblyGuide"); var active = state.steps[state.activeIndex]; if (!root || !active) return; var viewer = root.querySelector("[data-assembly-viewer]"); if (!viewer) return; var image = viewer.querySelector("[data-assembly-image]"); var section = viewer.querySelector("[data-assembly-section]"); var title = viewer.querySelector("[data-assembly-title]"); var progress = viewer.querySelector("[data-assembly-progress]"); var addendum = viewer.querySelector("[data-assembly-addendum]"); var addendumTitle = viewer.querySelector("[data-assembly-addendum-title]"); var prev = viewer.querySelector("[data-assembly-prev]"); var next = viewer.querySelector("[data-assembly-next]"); image.src = active.imageUrl; image.alt = active.isAddendum ? text(active.step.title) + " detail for " + text(active.parentTitle) : text(active.step.title); section.textContent = text(active.section.title); title.textContent = active.isAddendum ? text(active.parentTitle) : text(active.step.title); progress.textContent = String(state.activeIndex + 1) + " / " + String(state.steps.length); addendum.hidden = !active.isAddendum; addendumTitle.hidden = !active.isAddendum; addendumTitle.textContent = active.isAddendum ? text(active.step.title) : ""; prev.disabled = state.activeIndex === 0; next.disabled = state.activeIndex === state.steps.length - 1; root.querySelectorAll("[data-assembly-jump]").forEach(function(button) { var isActive = Number(button.dataset.assemblyJump) === state.activeIndex; button.classList.toggle("is-active", isActive); button.setAttribute("aria-current", isActive ? "step" : "false"); }); } function createViewer() { var viewer = createElement("div", "assembly-viewer instruction-stage"); viewer.setAttribute("data-assembly-viewer", "true"); var top = createElement("div", "assembly-viewer-top"); var meta = createElement("div", "assembly-viewer-meta"); var titleRow = createElement("div", "assembly-current-title-row"); var title = createElement("h3", "", ""); title.setAttribute("data-assembly-title", "true"); titleRow.appendChild(title); meta.appendChild(titleRow); var flags = createElement("div", "assembly-current-flags"); var progress = createElement("span", "assembly-progress", ""); progress.setAttribute("data-assembly-progress", "true"); var addendum = createElement("span", "assembly-step-kind", "Addendum"); addendum.setAttribute("data-assembly-addendum", "true"); addendum.hidden = true; flags.appendChild(progress); flags.appendChild(addendum); meta.appendChild(flags); var controls = createElement("div", "assembly-viewer-controls"); var prev = createButton("Previous", "button secondary assembly-nav-button", function() { moveStep(-1); }); prev.setAttribute("data-assembly-prev", "true"); var next = createButton("Next", "button assembly-nav-button", function() { moveStep(1); }); next.setAttribute("data-assembly-next", "true"); controls.appendChild(prev); controls.appendChild(next); top.appendChild(meta); top.appendChild(controls); var stage = createElement("div", "assembly-image-stage"); var image = document.createElement("img"); image.setAttribute("data-assembly-image", "true"); image.loading = "eager"; stage.appendChild(image); var caption = createElement("div", "assembly-stage-caption"); caption.appendChild(createElement("div", "assembly-current-section", "")); caption.lastChild.setAttribute("data-assembly-section", "true"); var addendumTitle = createElement("div", "assembly-addendum-title", ""); addendumTitle.setAttribute("data-assembly-addendum-title", "true"); addendumTitle.hidden = true; caption.appendChild(addendumTitle); viewer.appendChild(top); viewer.appendChild(stage); viewer.appendChild(caption); return viewer; } function createReference() { if (!state.reference) return null; var reference = createElement("details", "assembly-reference"); var summary = createElement("summary", "", state.reference.title + " Reference"); reference.appendChild(summary); var image = document.createElement("img"); image.src = state.reference.imageUrl; image.alt = state.reference.title; image.loading = "lazy"; reference.appendChild(image); return reference; } function createContents() { var details = createElement("details", "assembly-contents"); var summary = createElement("summary", "", "Jump to section"); details.appendChild(summary); var list = createElement("div", "assembly-contents-list"); state.steps.forEach(function(item, index) { var label = item.isAddendum ? "Detail: " + text(item.step.title) : text(item.step.title); var button = createButton(label, "assembly-contents-item" + (item.isAddendum ? " is-addendum" : ""), function() { setActiveStep(index); details.open = false; }); button.setAttribute("data-assembly-jump", String(index)); var section = createElement( "span", "assembly-contents-section", item.isAddendum ? "For " + item.parentTitle : item.section.title ); button.appendChild(section); list.appendChild(button); }); details.appendChild(list); return details; } function renderAssemblyGuide() { var root = document.getElementById("assemblyGuide"); if (!root) return; var manifest = getAssemblyManifest(); state.steps = flattenManifest(manifest); state.activeIndex = 0; state.reference = getReference(manifest); root.innerHTML = ""; if (!state.steps.length || !manifest.imageBase) { root.appendChild(createElement("div", "card fine-print", "Assembly guide is not configured.")); return; } var viewer = createViewer(); root.appendChild(viewer); var top = viewer.querySelector(".assembly-viewer-top"); if (top) top.appendChild(createContents()); var tools = createElement("div", "assembly-stage-tools"); var reference = createReference(); if (reference) tools.appendChild(reference); if (tools.childNodes.length) viewer.appendChild(tools); renderActiveStep(); } window.renderAssemblyGuide = renderAssemblyGuide; })();
